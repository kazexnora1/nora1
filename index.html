/* ============================================================ JCL9Match バックエンド（Google Apps Script・APIのみ） ------------------------------------------------------------ このスクリプトはデータ保存だけを行う API です。 画面（フロント）は GitHub Pages などの完全公開サイトに置き、 そこから fetch() でこの Web アプリの URL を呼び出します。 使用シート（存在しなければ自動作成されます）: - Roster : id | name | fargo - Results : timestamp | player1 | player2 | score1 | score2 | raceTo1 | raceTo2 | winner デプロイ方法: 1. このコードを Code.gs に貼り付けて保存 2. スプレッドシートのURLからIDをコピーし、下の SPREADSHEET_ID に貼る 3. デプロイ → 新しいデプロイ → 種類「ウェブアプリ」 実行するユーザー：自分 / アクセス：全員 4. 発行された .../exec の URL をコピー 5. フロント（index.html）の GAS_URL にそのURLを貼る ============================================================ */ const ROSTER_SHEET = "Roster"; const RESULTS_SHEET = "Results"; // スプレッドシートのURLの https://docs.google.com/spreadsheets/d/【ここ】/edit の部分をコピーして貼る const SPREADSHEET_ID = "1-N6y43jbI5i4dlid4B1cfC0J3C1fCTX0wWwu4PolBc8"; function getSpreadsheet_() { if (SPREADSHEET_ID) return SpreadsheetApp.openById(SPREADSHEET_ID); const ss = SpreadsheetApp.getActiveSpreadsheet(); if (!ss) throw new Error("スプレッドシートが見つかりません。SPREADSHEET_ID を設定してください。"); return ss; } function getSheet_(name, headers) { const ss = getSpreadsheet_(); let sh = ss.getSheetByName(name); if (!sh) { sh = ss.insertSheet(name); sh.appendRow(headers); } return sh; } function respond_(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } /* ---------- ロスター ---------- */ function loadRoster_() { const sh = getSheet_(ROSTER_SHEET, ["id", "name", "fargo"]); const rows = sh.getDataRange().getValues(); const out = []; for (let i = 1; i < rows.length; i++) { const [id, name, fargo] = rows[i]; if (!name && !id) continue; out.push({ id: Number(id) || i, name: String(name || ""), r: fargo === "" || fargo == null ? "" : String(fargo), }); } return out; } function saveRoster_(roster) { const sh = getSheet_(ROSTER_SHEET, ["id", "name", "fargo"]); sh.clearContents(); sh.appendRow(["id", "name", "fargo"]); (roster || []).forEach((p) => sh.appendRow([p.id, p.name, p.r])); } /* ---------- 試合結果 ---------- */ function loadResults_() { const sh = getSheet_(RESULTS_SHEET, [ "timestamp", "player1", "player2", "score1", "score2", "raceTo1", "raceTo2", "winner", ]); const rows = sh.getDataRange().getValues(); const out = []; for (let i = 1; i < rows.length; i++) { const [ts, p1, p2, s1, s2, r1, r2, w] = rows[i]; if (!p1 && !p2) continue; out.push({ finishedAt: ts instanceof Date ? ts.toISOString() : String(ts), names: [String(p1 || ""), String(p2 || "")], score: [Number(s1) || 0, Number(s2) || 0], raceTo: [Number(r1) || 0, Number(r2) || 0], winner: Number(w), }); } return out; } function addResult_(result) { const sh = getSheet_(RESULTS_SHEET, [ "timestamp", "player1", "player2", "score1", "score2", "raceTo1", "raceTo2", "winner", ]); sh.appendRow([ result.finishedAt || new Date().toISOString(), (result.names && result.names[0]) || "", (result.names && result.names[1]) || "", (result.score && result.score[0]) || 0, (result.score && result.score[1]) || 0, (result.raceTo && result.raceTo[0]) || 0, (result.raceTo && result.raceTo[1]) || 0, result.winner, ]); } /* ---------- エントリポイント（JSON API） ---------- */ function doGet(e) { const action = (e && e.parameter && e.parameter.action) || null; if (action === "roster") return respond_(loadRoster_()); if (action === "results") return respond_(loadResults_()); if (action === "saveRoster") { try { const roster = JSON.parse(e.parameter.data || "[]"); saveRoster_(roster); return respond_({ ok: true }); } catch (err) { return respond_({ error: "invalid data: " + err.message }); } } if (action === "addResult") { try { const result = JSON.parse(e.parameter.data || "{}"); addResult_(result); return respond_({ ok: true }); } catch (err) { return respond_({ error: "invalid data: " + err.message }); } } return respond_({ error: "unknown action: " + action }); } function doPost(e) { let body; try { body = JSON.parse((e && e.postData && e.postData.contents) || "{}"); } catch (err) { return respond_({ error: "invalid JSON body" }); } if (body.action === "saveRoster") { saveRoster_(body.roster); return respond_({ ok: true }); } if (body.action === "addResult") { addResult_(body.result); return respond_({ ok: true }); } return respond_({ error: "unknown action: " + body.action }); }
